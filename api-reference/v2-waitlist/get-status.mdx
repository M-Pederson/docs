<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>

<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>

### Query Parameters

<ParamField query="email" type="string" required>
  Email address to check.
</ParamField>

### Response

<ResponseField name="email" type="string">
  The email address checked.
</ResponseField>

<ResponseField name="position" type="string">
  Current position in the waitlist.
</ResponseField>

<ResponseField name="people_ahead" type="string">
  Number of people ahead in the waitlist.
</ResponseField>

<ResponseField name="status" type="string">
  Current status (e.g. `pending`, `invited`, `converted`).
</ResponseField>

<ResponseField name="referral_code" type="string">
  The user's referral code.
</ResponseField>

<ResponseField name="referral_link" type="string">
  The user's unique referral link.
</ResponseField>

<ResponseField name="referral_count" type="string">
  Number of successful referrals.
</ResponseField>

<ResponseField name="link_clicks" type="string">
  Number of clicks on the referral link.
</ResponseField>

<ResponseField name="joined_at" type="string">
  Timestamp when the user joined the waitlist.
</ResponseField>
