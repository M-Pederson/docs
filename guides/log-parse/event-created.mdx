## calendar-webhook-parse-event — #2477

Triggered by the Nylas `event.created` webhook when a new calendar event is received. Logs the raw event payload to `log_webhook_nylas`, validates the grant and calendar exist (early return if not), handles recurring events by fetching or creating `master_event` records via the Nylas API (with null-safe conferencing and all-day event support), creates the `event` record, then iterates through participants — determining role (organizer type 5 vs. participant type 6) up front, resolving each to a `master_person` (creating one from the email address if needed), copying to `my_person`, and logging a `calendar_event_activity`. Organizer timestamp falls back to `created_at` when `updated_at` is null. Participants are iterated directly without duplicating the organizer. Kicks off async primary email resolution for each participant.

````xanoscript
function "mvp/nylas-webhook/calendar-webhook-parse-event" {
  input {
    text grant_id filters=trim
    json event
  }

  stack {
    db.add log_webhook_nylas {
      data = {
        created_at: "now"
        grant_id  : $input.grant_id
        type      : "event"
        webhook   : "calendar-webhook-parse-event"
        event_body: $input.event
      }
    } as $log

    // DEV GATE — only executes on live datasource
    group {
      stack {
        conditional {
          if ($env.$datasource != "live") {
            return {
              value = ""
            }
          }

          else {
            !return {
              value = ""
            }
          }
        }
      }
    }

    try_catch {
      try {
        var $created {
          value = false
        }

        function.run "mvp/nylas/get-nylas-keys" as $keys
        db.get nylas_grant {
          field_name = "grant_id"
          field_value = $input.grant_id
        } as $nylas_grant

        // Guard — grant must exist
        conditional {
          if ($nylas_grant == null) {
            return {
              value = "Grant not found for grant_id"
            }
          }
        }

        db.get calendar {
          field_name = "external_id"
          field_value = $input.event.calendar_id
          output = ["id", "external_id"]
        } as $calendar

        // Calendar must exist (checked before event exists query)
        conditional {
          if ($calendar == null) {
            return {
              value = "Calendar does not exist"
            }
          }
        }

        db.query event {
          where = $db.event.external_id == $input.event.id && $db.event.calendar_id == $calendar.id
          return = {type: "exists"}
        } as $hasEvent

        conditional {
          if ($hasEvent) {
            return {
              value = "Event already exists"
            }
          }
        }

        // Handle master event (recurring)
        conditional {
          if ($input.event|has:"master_event_id") {
            db.get master_event {
              field_name = "external_id"
              field_value = $input.event.master_event_id
              output = ["id"]
            } as $master_event

            conditional {
              if ($master_event == null) {
                api.request {
                  url = "https://api.us.nylas.com/v3/grants/%s/events/%s"
                    |sprintf:$input.grant_id:$input.event.master_event_id
                  method = "GET"
                  params = {}
                    |set:"calendar_id":$calendar.external_id
                  headers = []
                    |push:"Accept: application/json"
                    |push:("Authorization: Bearer %s"|sprintf:$keys.nylas_api_key)
                    |push:"Content-Type: application/json"
                } as $nylas_master_event_response

                // Null-safe conferencing + all-day event handling
                db.add master_event {
                  data = {
                    created_at : "now"
                    external_id: $nylas_master_event_response.response.result.data.id
                    calendar_id: $calendar.id
                    recurrence : $nylas_master_event_response.response.result.data.recurrence[0] ?: null
                    title      : $nylas_master_event_response.response.result.data.title
                    start      : $nylas_master_event_response.response.result.data.when.start_time ? ($nylas_master_event_response.response.result.data.when.start_time|multiply:1000) : $nylas_master_event_response.response.result.data.when.start_date
                    end        : $nylas_master_event_response.response.result.data.when.end_time ? ($nylas_master_event_response.response.result.data.when.end_time|multiply:1000) : $nylas_master_event_response.response.result.data.when.end_date
                    provider   : $nylas_master_event_response.response.result.data.conferencing.provider ?: null
                    url        : $nylas_master_event_response.response.result.data.conferencing.details.url ?: null
                  }

                  output = ["id"]
                } as $master_event
              }
            }
          }

          else {
            var $master_event {
              value = {}|set:"id":null
            }
          }
        }

        var $participant_count {
          value = $input.event.participants|count
        }

        // Create the EVENT record
        db.add event {
          data = {
            created_at     : "now"
            updated_at     : null
            title          : $input.event.title
            description    : $input.event|get:"description":null
            external_id    : $input.event.id
            start          : $input.event.when.start_time ? ($input.event.when.start_time|multiply:1000) : $input.event.when.start_date
            end            : $input.event.when.end_time ? ($input.event.when.end_time|multiply:1000) : $input.event.when.end_date
            master_event_id: $master_event.id
            calendar_id    : $calendar.id
            provider       : $input.event.conferencing.provider ?: null
            url            : $input.event.conferencing.details.url ?: null
            organize       : false
            event_data     : $input.event
            deleted_at     : null
          }
        } as $event_db

        var $dataSourceId {
          value = 21
        }

        // Guard organizer existence before accessing email
        var $organizer_email {
          value = ($input.event|has:"organizer") ? ($input.event.organizer.email|to_lower) : null
        }

        // Iterate participants directly — no organizer push (avoids duplicates)
        // Role variables set up front, then single identity/activity flow
        foreach ($input.event.participants) {
          each as $participant {
            // Determine role: organizer (type 5) vs participant (type 6)
            conditional {
              if ($organizer_email != null && ($participant.email|to_lower) == $organizer_email) {
                var $participantType {
                  value = 5
                }

                var $cardDescriptor {
                  value = "You scheduled a meeting"
                }

                // Fall back to created_at when updated_at is null
                var $activityTimestamp {
                  value = ($input.event|get:"updated_at":null) ? ($input.event.updated_at|to_int|multiply:1000) : ($input.event.created_at|to_int|multiply:1000)
                }
              }

              else {
                var $participantType {
                  value = 6
                }

                var $cardDescriptor {
                  value = "Participant in a scheduled meeting"
                }

                var $activityTimestamp {
                  value = $input.event.created_at|to_int|multiply:1000
                }
              }
            }

            function.run "mvp/identity/check-identity-email" {
              input = {
                email_address: $participant.email
                user_id      : $nylas_grant.user_id
              }
            } as $identityEmail

            conditional {
              if ($identityEmail.master_person_id != null) {
                // Existing master person
                group {
                  stack {
                    db.get master_person {
                      field_name = "id"
                      field_value = $identityEmail.master_person_id
                      output = ["id", "name"]
                    } as $masterPerson

                    function.run "mvp/my/copy-to-my-person" {
                      input = {
                        master_person_id: $masterPerson.id
                        user_id         : $nylas_grant.user_id
                        email_address   : ""
                      }
                    } as $copyToMy

                    db.add calendar_event_activity {
                      data = {
                        created_at        : "now"
                        user_id           : $nylas_grant.user_id
                        my_person_id      : $copyToMy.my_person_id
                        activity_type_id  : $participantType
                        subject           : $input.event|get:"title":null
                        card_descriptor   : $cardDescriptor
                        activity_timestamp: $activityTimestamp
                        grant_id          : $input.event|get:"grant_id":null
                        event_id          : $event_db.id
                        interlocutor_email: $identityEmail.master_email_id
                        participant_count : $participant_count
                        data_source_id    : $dataSourceId
                      }
                    } as $activity

                    db.edit my_person {
                      field_name = "id"
                      field_value = $copyToMy.my_person_id
                      data = {
                        updated_at                     : now
                        last_email_activity_id         : null
                        last_calendar_event_activity_id: $activity.id
                        last_activity_at               : $activityTimestamp
                      }

                      output = ["id"]
                    } as $myPerson

                    function.run "mvp/email/set-primary-email_v2" {
                      runtime_mode = "async-shared"
                      input = {
                        email_address: $participant.email
                        user_id      : $nylas_grant.user_id
                      }
                    }
                  }
                }
              }

              else {
                // New master person — create identity + person
                group {
                  stack {
                    var $tempName {
                      value = $participant.email
                        |split:"@"
                        |first
                        |replace:".":" "
                        |capitalize
                    }

                    group {
                      stack {
                        function.run "mvp/format/name-format" {
                          input = {full_name: $tempName}
                        } as $nameFormat

                        function.run "mvp/get-add/master-person" {
                          input = {
                            first_name    : $nameFormat.first_name
                            last_name     : $nameFormat.last_name
                            links         : []
                            phone_numbers : []
                            email         : []|push:($participant.email|to_lower)
                            company_name  : null
                            title         : null
                            data_source_id: $dataSourceId
                            user_id       : $nylas_grant.user_id
                            full_name     : $nameFormat.full_name
                            queue         : false
                          }
                        } as $getIdentity
                      }
                    }

                    db.get master_person {
                      field_name = "id"
                      field_value = $getIdentity.master_person_id
                      output = [
                        "id"
                        "name"
                        "first_name"
                        "middle_name"
                        "last_name"
                        "suffix"
                        "nickname"
                        "birthdate"
                        "status_orbiter"
                        "email_signature_parsed"
                      ]
                    } as $masterPerson

                    function.run "mvp/my/copy-to-my-person" {
                      input = {
                        master_person_id: $masterPerson.id
                        user_id         : $nylas_grant.user_id
                        email_address   : ""
                      }
                    } as $copyToMy

                    group {
                      stack {
                        db.get master_email {
                          field_name = "email_address"
                          field_value = $participant.email
                          output = ["id"]
                        } as $masterEmail

                        db.add calendar_event_activity {
                          data = {
                            created_at        : "now"
                            user_id           : $nylas_grant.user_id
                            my_person_id      : $copyToMy.my_person_id
                            activity_type_id  : $participantType
                            subject           : $input.event|get:"title":null
                            card_descriptor   : $cardDescriptor
                            activity_timestamp: $activityTimestamp
                            grant_id          : $input.event|get:"grant_id":null
                            event_id          : $event_db.id
                            interlocutor_email: $masterEmail.id
                            participant_count : $participant_count
                            data_source_id    : $dataSourceId
                          }
                        } as $activity

                        db.edit my_person {
                          field_name = "id"
                          field_value = $copyToMy.my_person_id
                          data = {
                            updated_at                     : now
                            last_email_activity_id         : null
                            last_calendar_event_activity_id: $activity.id
                            last_activity_at               : $activityTimestamp
                          }
                        }

                        function.run "mvp/email/set-primary-email_v2" {
                          runtime_mode = "async-shared"
                          input = {
                            email_address: $participant.email
                            user_id      : $nylas_grant.user_id
                          }
                        }
                      }
                    }
                  }
                }
              }
            }
          }
        }
      }

      catch {
        db.edit log_webhook_nylas {
          field_name = "id"
          field_value = $log.id
          data = {error: $error.message}
        } as $log
      }
    }
  }

  response = "success"
  tags = ["event.created webhook", "requires-nylas-auth", "in progress"]
  history = 100
}
````

### Sub-functions called

| Function | Description |
|----------|-------------|
| `get-nylas-keys` | Retrieves Nylas API credentials |
| `check-identity-email` | Resolves an email address to a master_person_id |
| `copy-to-my-person` | Copies master_person to user's my_person table |
| `name-format` | Parses a full name into first/middle/last/suffix |
| `master-person` (get-add) | Gets or creates a master_person record |
| `set-primary-email_v2` | (async) Sets primary email for a person |
