---

<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>
<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>
<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

<ResponseField name="id" type="integer">
  The email integration record ID.
</ResponseField>
<ResponseField name="timerange" type="integer">
  The time range in days for email parsing (default: 30).
</ResponseField>
<ResponseField name="email_account" type="string">
  The connected email address.
</ResponseField>
<ResponseField name="provider" type="string">
  The email provider. One of: `google`, `microsoft`, `icloud`, `imap`, `yahoo`.
</ResponseField>
