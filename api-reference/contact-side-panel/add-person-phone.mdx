<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>
<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>
<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

<ParamField body="my_person_id" type="integer">
  The person ID to add the phone to.
</ParamField>
<ParamField body="phone_number" type="string">
  The phone number.
</ParamField>
<ParamField body="phone_extension" type="integer">
  Phone extension.
</ParamField>
<ParamField body="phone_type" type="string">
  Type: `mobile`, `office`, `other`, or `home`. Defaults to `mobile`.
</ParamField>
<ParamField body="phone_note" type="string">
  Optional note about the phone.
</ParamField>

<ResponseField name="id" type="integer">Record ID.</ResponseField>
<ResponseField name="my_person_id" type="integer">Person ID.</ResponseField>
<ResponseField name="master_phone_id" type="integer">Master phone record ID.</ResponseField>
