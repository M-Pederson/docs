---
title: 'List Master Persons'
api: 'GET https://xh2o-yths-38lt.n7c.xano.io/api:WKgay2AU/master-persons'
description: 'Query all master person records'
---

## Overview

Returns a paginated list of all master person records, with optional search filtering.

## Headers

<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>

<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

## Query Parameters

<ParamField query="query" type="string">
  Search query to filter persons.
</ParamField>

<ParamField query="page" type="integer">
  Page number for pagination.
</ParamField>

<ParamField query="limit" type="integer">
  Number of items per page.
</ParamField>

<ParamField query="offset" type="integer">
  Number of items to skip.
</ParamField>

## Response

<ResponseField name="itemsReceived" type="integer">
  Number of items returned in the current response.
</ResponseField>

<ResponseField name="curPage" type="integer">
  The current page number.
</ResponseField>

<ResponseField name="nextPage" type="integer">
  The next page number, or null if no more pages.
</ResponseField>

<ResponseField name="prevPage" type="integer">
  The previous page number, or null if on the first page.
</ResponseField>

<ResponseField name="offset" type="integer">
  The offset used for the query.
</ResponseField>

<ResponseField name="perPage" type="integer">
  Number of items per page.
</ResponseField>

<ResponseField name="itemsTotal" type="integer">
  Total number of items matching the query.
</ResponseField>

<ResponseField name="pageTotal" type="integer">
  Total number of pages.
</ResponseField>

<ResponseField name="items" type="array">
  Array of master person records containing id, orbiter_user, name, avatar, master_company_id, current_title, primary_location_id, and bio.
</ResponseField>
