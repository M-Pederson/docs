---
title: "Create Important Date"
api: "POST https://xh2o-yths-38lt.n7c.xano.io/api:mGnfV_UB/important-dates"
description: "Create a new important date for a person."
---

<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>
<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>
<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

<ParamField body="master_person_id" type="integer">
  The ID of the master person.
</ParamField>
<ParamField body="date_descriptor_id" type="integer">
  The ID of the date descriptor.
</ParamField>
<ParamField body="annual_recurring" type="boolean">
  Whether this date recurs annually.
</ParamField>
<ParamField body="date" type="string" required>
  The date in `YYYY-MM-DD` format.
</ParamField>
<ParamField body="note" type="string">
  An optional note about the date.
</ParamField>
<ParamField body="reminder" type="integer">
  Number of days before the date to send a reminder.
</ParamField>
<ParamField body="date_descriptor" type="string">
  A text descriptor for the date (alternative to `date_descriptor_id`).
</ParamField>

<ResponseField name="id" type="integer">
  The created important date record ID.
</ResponseField>
<ResponseField name="created_at" type="number">
  Creation timestamp.
</ResponseField>
<ResponseField name="my_person_id" type="integer">
  The associated person ID.
</ResponseField>
<ResponseField name="date_descriptor_id" type="integer">
  The date descriptor ID.
</ResponseField>
<ResponseField name="annual_recurring" type="boolean">
  Whether the date recurs annually.
</ResponseField>
<ResponseField name="date" type="string">
  The date value.
</ResponseField>
<ResponseField name="note" type="string">
  The note.
</ResponseField>
<ResponseField name="reminder" type="integer">
  Days before to remind.
</ResponseField>
