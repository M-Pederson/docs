## mvp/edges/create-work-edges — #2800

Creates all work-related edges between a person and company. Supports multiple edge types per company based on title detection.

```text
mvp/edges/create-work-edges — #2800
```

```cypher
// Version v1
WITH {{$var.cypher}} AS g

MERGE (p:Person {uuid: g.person_uuid})
SET p.name = CASE WHEN g.person_name IS NOT NULL THEN g.person_name ELSE p.name END

MERGE (c:Company {uuid: g.company_uuid})
SET c.name = CASE WHEN g.company_name IS NOT NULL THEN g.company_name ELSE c.name END

WITH p, c, g

FOREACH (x IN CASE WHEN g.is_founder = true THEN [1] ELSE [] END |
  MERGE (p)-[r:FOUNDED]->(c)
  SET r.title = g.founder_title,
      r.weight = 5,
      r.description = CASE
        WHEN toLower(g.founder_title) CONTAINS 'co-founder'
          OR toLower(g.founder_title) CONTAINS 'cofounder'
        THEN CASE
          WHEN g.founder_start_year IS NOT NULL
          THEN p.name + ' co-founded ' + c.name + ' in ' + toString(g.founder_start_year) + '.'
          ELSE p.name + ' is the Co-Founder of ' + c.name + '.'
        END
        ELSE CASE
          WHEN g.founder_start_year IS NOT NULL
          THEN p.name + ' founded ' + c.name + ' in ' + toString(g.founder_start_year) + '.'
          ELSE p.name + ' is the Founder of ' + c.name + '.'
        END
      END,
      r.valid_from = g.founder_valid_from,
      r.valid_to = g.founder_valid_to,
      r.uuid = coalesce(r.uuid, randomUUID())
)

FOREACH (x IN CASE WHEN g.is_board_member = true THEN [1] ELSE [] END |
  MERGE (p)-[r:BOARD_MEMBER_OF]->(c)
  SET r.title = g.board_title,
      r.weight = 8,
      r.description = g.board_description,
      r.valid_from = g.board_valid_from,
      r.valid_to = g.board_valid_to,
      r.uuid = coalesce(r.uuid, randomUUID())
)

FOREACH (x IN CASE WHEN g.is_advisor = true THEN [1] ELSE [] END |
  MERGE (p)-[r:ADVISOR_TO]->(c)
  SET r.title = g.advisor_title,
      r.weight = 12,
      r.description = g.advisor_description,
      r.valid_from = g.advisor_valid_from,
      r.valid_to = g.advisor_valid_to,
      r.uuid = coalesce(r.uuid, randomUUID())
)

FOREACH (x IN CASE WHEN g.is_investor = true THEN [1] ELSE [] END |
  MERGE (p)-[r:INVESTED_IN]->(c)
  SET r.title = g.investor_title,
      r.weight = 15,
      r.description = g.investor_description,
      r.valid_from = g.investor_valid_from,
      r.valid_to = g.investor_valid_to,
      r.uuid = coalesce(r.uuid, randomUUID())
)

FOREACH (x IN CASE WHEN g.is_member = true THEN [1] ELSE [] END |
  MERGE (p)-[r:MEMBER_OF]->(c)
  SET r.title = g.member_title,
      r.weight = 20,
      r.description = g.member_description,
      r.valid_from = g.member_valid_from,
      r.valid_to = g.member_valid_to,
      r.uuid = coalesce(r.uuid, randomUUID())
)

FOREACH (x IN CASE WHEN g.is_works_at = true THEN [1] ELSE [] END |
  MERGE (p)-[r:WORKS_AT]->(c)
  SET r.title = g.works_at_title,
      r.weight = g.works_at_weight,
      r.seniority_level = g.works_at_seniority,
      r.valid_from = g.valid_from,
      r.valid_to = g.valid_to,
      r.total_roles = g.total_roles,
      r.role_history = g.role_history,
      r.uuid = coalesce(r.uuid, randomUUID())
)

WITH p, c, g

OPTIONAL MATCH (p)-[founded:FOUNDED]->(c)
OPTIONAL MATCH (p)-[board:BOARD_MEMBER_OF]->(c)
OPTIONAL MATCH (p)-[advisor:ADVISOR_TO]->(c)
OPTIONAL MATCH (p)-[investor:INVESTED_IN]->(c)
OPTIONAL MATCH (p)-[member:MEMBER_OF]->(c)
OPTIONAL MATCH (p)-[works:WORKS_AT]->(c)

RETURN p.name AS person,
       p.uuid AS person_uuid,
       c.name AS company,
       c.uuid AS company_uuid,
       g.is_founder AS created_founded,
       founded.uuid AS founded_edge_uuid,
       g.is_board_member AS created_board_member,
       board.uuid AS board_member_edge_uuid,
       g.is_advisor AS created_advisor,
       advisor.uuid AS advisor_edge_uuid,
       g.is_investor AS created_investor,
       investor.uuid AS investor_edge_uuid,
       g.is_member AS created_member,
       member.uuid AS member_edge_uuid,
       g.is_works_at AS created_works_at,
       works.uuid AS works_at_edge_uuid
```

## mvp/edges/create-education-edges — #1898

Creates ATTENDED edge (one per school, collapsed from multiple records) plus derived ATTENDED_TOGETHER and STUDIED_UNDER relationships.

```text
mvp/edges/create-education-edges — #1898
```

```cypher
// Version v1.1 — FalkorDB-compat: date().year replaced by precomputed {{$var.currentYear}}
MATCH (person {uuid: "{{$var.firstRecord._master_person.node_uuid}}"})
MATCH (school {uuid: "{{$var.firstRecord._master_company.node_uuid}}"})
MERGE (person)-[r:ATTENDED]->(school)
ON CREATE SET r.uuid = randomUUID()

WITH person, r, school,
     "{{$var.descrip}}" AS descripStr,
     "{{$var.fieldOfStudyStr}}" AS fieldOfStudyStr,
     "{{$var.degreeStr}}" AS degreeStr,
     "{{$var.activityStr}}" AS activitiesStr,
     "{{$var.validFromStr}}" AS validFromStr,
     "{{$var.validToStr}}" AS validToStr

FOREACH (x IN CASE WHEN descripStr <> "" THEN [1] ELSE [] END |
  SET r.description = descripStr
)
FOREACH (x IN CASE WHEN fieldOfStudyStr <> "" THEN [1] ELSE [] END |
  SET r.field_of_study = fieldOfStudyStr
)
FOREACH (x IN CASE WHEN degreeStr <> "" THEN [1] ELSE [] END |
  SET r.degree = degreeStr
)
FOREACH (x IN CASE WHEN activitiesStr <> "" THEN [1] ELSE [] END |
  SET r.activities_societies = activitiesStr
)
FOREACH (x IN CASE WHEN validFromStr <> "" THEN [1] ELSE [] END |
  SET r.valid_from = validFromStr
)
FOREACH (x IN CASE WHEN validToStr <> "" AND toInteger(substring(validToStr, 0, 4)) <= {{$var.currentYear}} THEN [1] ELSE [] END |
  SET r.valid_to = validToStr
)

SET r.weight = 40

WITH person, r, school, r.uuid AS edge_uuid

-- Derived: ATTENDED_TOGETHER
OPTIONAL MATCH (person)-[myAttended:ATTENDED]->(school)<-[theirAttended:ATTENDED]-(peer:Person)
WHERE person <> peer
  AND myAttended.valid_from IS NOT NULL
  AND myAttended.valid_to IS NOT NULL
  AND theirAttended.valid_from IS NOT NULL
  AND theirAttended.valid_to IS NOT NULL
  AND myAttended.valid_from <= theirAttended.valid_to
  AND myAttended.valid_to >= theirAttended.valid_from

FOREACH (x IN CASE WHEN peer IS NOT NULL THEN [1] ELSE [] END |
  MERGE (person)-[at:ATTENDED_TOGETHER]-(peer)
  ON CREATE SET at.uuid = randomUUID()
  SET at.school = school.name,
      at.overlap_start = CASE
        WHEN myAttended.valid_from > theirAttended.valid_from
        THEN myAttended.valid_from ELSE theirAttended.valid_from END,
      at.overlap_end = CASE
        WHEN myAttended.valid_to < theirAttended.valid_to
        THEN myAttended.valid_to ELSE theirAttended.valid_to END,
      at.valid_from = CASE
        WHEN myAttended.valid_from > theirAttended.valid_from
        THEN myAttended.valid_from ELSE theirAttended.valid_from END,
      at.valid_to = CASE
        WHEN myAttended.valid_to < theirAttended.valid_to
        THEN myAttended.valid_to ELSE theirAttended.valid_to END,
      at.same_degree = CASE
        WHEN myAttended.degree IS NOT NULL AND theirAttended.degree IS NOT NULL
         AND myAttended.degree = theirAttended.degree THEN true ELSE false END,
      at.same_major = CASE
        WHEN myAttended.major IS NOT NULL AND theirAttended.major IS NOT NULL
         AND myAttended.major = theirAttended.major THEN true ELSE false END,
      at.same_field = CASE
        WHEN myAttended.field IS NOT NULL AND theirAttended.field IS NOT NULL
         AND myAttended.field = theirAttended.field THEN true ELSE false END,
      at.relationship_strength = CASE
        WHEN myAttended.major IS NOT NULL AND theirAttended.major IS NOT NULL
         AND myAttended.major = theirAttended.major THEN 3
        WHEN myAttended.field IS NOT NULL AND theirAttended.field IS NOT NULL
         AND myAttended.field = theirAttended.field THEN 2
        ELSE 1 END,
      at.weight = CASE
        WHEN myAttended.major IS NOT NULL AND theirAttended.major IS NOT NULL
         AND myAttended.major = theirAttended.major THEN 35
        WHEN myAttended.field IS NOT NULL AND theirAttended.field IS NOT NULL
         AND myAttended.field = theirAttended.field THEN 50
        ELSE 70 END,
      at.context = CASE
        WHEN myAttended.major IS NOT NULL AND theirAttended.major IS NOT NULL
         AND myAttended.major = theirAttended.major
        THEN "Same major (" + myAttended.major + ")"
        WHEN myAttended.field IS NOT NULL AND theirAttended.field IS NOT NULL
         AND myAttended.field = theirAttended.field
        THEN "Same field (" + toString(myAttended.field) + ")"
        ELSE "Overlapped at " + school.name END,
      at.description = person.name + " and " + peer.name +
        " studied together at " + school.name
)

WITH DISTINCT person, school, edge_uuid

-- Derived: STUDIED_UNDER
OPTIONAL MATCH (person)-[myAttended:ATTENDED]->(school)<-[taught:TAUGHT_AT]-(instructor:Person)
WHERE myAttended.valid_from IS NOT NULL
  AND myAttended.valid_to IS NOT NULL
  AND taught.valid_from IS NOT NULL
  AND taught.valid_to IS NOT NULL
  AND myAttended.valid_from <= taught.valid_to
  AND myAttended.valid_to >= taught.valid_from
  AND (myAttended.field IS NULL OR taught.department IS NULL
       OR myAttended.field = taught.department)

FOREACH (x IN CASE WHEN instructor IS NOT NULL THEN [1] ELSE [] END |
  MERGE (person)-[su:STUDIED_UNDER]->(instructor)
  ON CREATE SET su.uuid = randomUUID()
  SET su.school = school.name,
      su.department = taught.department,
      su.overlap_start = CASE
        WHEN myAttended.valid_from > taught.valid_from
        THEN myAttended.valid_from ELSE taught.valid_from END,
      su.overlap_end = CASE
        WHEN myAttended.valid_to < taught.valid_to
        THEN myAttended.valid_to ELSE taught.valid_to END,
      su.weight = 10,
      su.description = person.name + " studied under " +
        instructor.name + " at " + school.name
)

RETURN edge_uuid
```

```cypher
// Version v1
// ============================================
// Certification: {{$var.certName}}
// ============================================
// Create Certification Node with Entity label
MERGE (certification:Entity:Certification {name: "{{$var.certName}}"})
SET certification += {
{% if $var.item.issue_date and $var.item.issue_date != "" %}issue_date: "{{$var.item.issue_date}}",{% endif %}
{% if $var.item.description and $var.item.description != "" %}description: "{{$var.item.description}}",{% endif %}
{% if $var.item.org_name and $var.item.org_name != "" %}issued_by: "{{$var.item.org_name}}",{% endif %}
  uuid: CASE
    WHEN certification.uuid IS NULL
    THEN randomUUID()
    ELSE certification.uuid
  END
}
SET certification.name_embedding = vecf32({{$var.embeddingArray}})
WITH certification
// Link Person to Certification
// TEMPORAL CONVENTION: currency derived from valid_to IS NULL
MATCH (person:Person {uuid: "{{$var.masterPerson.node_uuid}}"})
MERGE (person)-[has_cert:HAS_CERTIFICATION]->(certification)
SET has_cert += {
  uuid: CASE
    WHEN has_cert.uuid IS NULL
    THEN randomUUID()
    ELSE has_cert.uuid
  END,
  weight: {% if $var.certName starts with 'Series' %}15{% else %}20{% endif %}
{% if $var.item.issue_date and $var.item.issue_date != "" %},valid_from: "{{$var.item.issue_date}}"{% endif %}
{% if $var.item.description and $var.item.description != "" %},description: "{{$var.item.description}}"{% endif %}
}
WITH person, certification, has_cert
// Match existing Company and add Organization label
MATCH (organization:Company {uuid: "{{$var.masterCompany.node_uuid}}"})
SET organization:Organization
WITH person, certification, has_cert, organization
// Link Certification to Organization
MERGE (certification)-[granted:GRANTED_BY]->(organization)
SET granted += {
  uuid: CASE
    WHEN granted.uuid IS NULL
    THEN randomUUID()
    ELSE granted.uuid
  END,
  weight: 10
}
WITH person, certification, has_cert, organization, granted
// Create Direct Person-Organization Certification Relationship
MERGE (person)-[certified:CERTIFIED_BY {cert_type: "{{$var.certName}}"}]->(organization)
ON CREATE SET
  certified.uuid = randomUUID(),
  certified.weight = 25,
  certified.certification_count = 1,
  certified.certifications = ["{{$var.certName}}"],
  certified.first_certified = {% if $var.item.issue_date and $var.item.issue_date != "" %}"{{$var.item.issue_date}}"{% else %}"unknown"{% endif %}

ON MATCH SET
  certified.weight = 25,
  certified.certification_count = certified.certification_count + 1,
  certified.certifications = CASE
    WHEN NOT "{{$var.certName}}" IN certified.certifications
    THEN certified.certifications + "{{$var.certName}}"
    ELSE certified.certifications
  END{% if $var.item.issue_date and $var.item.issue_date != "" %},
  certified.most_recent = CASE
    WHEN "{{$var.item.issue_date}}" > coalesce(certified.most_recent, "")
    THEN "{{$var.item.issue_date}}"
    ELSE certified.most_recent
  END{% endif %}

// Return all created/updated nodes and edges with their UUIDs
WITH person, certification, has_cert, organization, granted, certified
RETURN {
  node_uuid: certification.uuid,
  edge_uuid: certified.uuid
} AS result
```

### MERGE Location Edges (City + Region + Country)

```cypher
// Version v1
MATCH (person:Entity {uuid: "{{$var.masterPerson.node_uuid}}"})
MATCH (city:Entity {uuid: "{{$var.cityRecord.node_uuid}}"})
MATCH (region:Entity {uuid: "{{$var.regionRecord.node_uuid}}"})
MATCH (country:Entity {uuid: "{{$var.countryRecord.node_uuid}}"})

MERGE (person)-[relates1:LOCATED_IN]->(city)
  ON CREATE SET
    relates1.uuid = randomUuid(),
    relates1.fact = person.name + ' is located in ' + city.name,
    relates1.weight = 10,
    relates1.name = "LOCATED_IN",
    relates1.valid_from = "{{'now'|date('Y-m')}}"

WITH person, city, region, country

MERGE (city)-[relates2:IN_REGION]->(region)
  ON CREATE SET
    relates2.uuid = randomUuid(),
    relates2.fact = city.name + ' is a city located in ' + region.name,
    relates2.weight = 25,
    relates2.name = "IN_REGION"

MERGE (region)-[relates3:IN_COUNTRY]->(country)
  ON CREATE SET
    relates3.uuid = randomUuid(),
    relates3.fact = CASE
      WHEN country.name = 'United States'
        THEN region.name + ' is a state located in the United States'
      WHEN country.name ENDS WITH 's'
        THEN region.name + ' is a region located in the ' + country.name
      ELSE region.name + ' is a region located in ' + country.name
    END,
    relates3.weight = 50,
    relates3.name = "IN_COUNTRY"

WITH person, city, region, country

OPTIONAL MATCH (person)-[relates1:LOCATED_IN]->(city)
OPTIONAL MATCH (city)-[relates2:IN_REGION]->(region)
OPTIONAL MATCH (region)-[relates3:IN_COUNTRY]->(country)

RETURN [relates1.uuid, relates2.uuid, relates3.uuid] AS edge_ids
```
