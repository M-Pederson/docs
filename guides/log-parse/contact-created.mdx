## nylas-webhook/contact-created — API #813

Triggered by the Nylas `contact.created` webhook when a new contact is synced. Validates the webhook signature using HMAC-SHA256, then broadcasts the raw contact data to the `new-contacts-created` realtime channel. Does not call any downstream functions — the contact processing is handled by the frontend or other listeners.

<Accordion title="XanoScript — click to expand">
````xanoscript
// Nylas sends new contcacts created to this endpoint
query "nylas-webhook/contact-created" verb=POST {
  api_group = "Nylas"

  input {
    text challenge? filters=trim
  }

  stack {
    conditional {
      if ($input.challenge != null) {
        util.set_header {
          value = "Content-Type: text/html; charset=UTF-8"
          duplicates = "replace"
        }

        return {
          value = $input.challenge|escape
        }
      }
    }

    util.get_raw_input {
      encoding = "none"
      exclude_middleware = false
    } as $raw_body

    // Check data source from client id from nylas
    conditional {
      if ($raw_body.data.application_id == $env.nylas_client_id) {
        db.set_datasource {
          value = "staging"
        }
      }

      else {
        db.set_datasource {
          value = "live"
        }
      }
    }

    function.run "mvp/nylas/get-nylas-keys" as $keys
    var $hash {
      value = $raw_body
        |hmac_sha256:$keys.nylas_webhook_contact_create_secret:false
    }

    conditional {
      if ($hash != $env.$http_headers["X-Nylas-Signature"]) {
        return {
          value = "signature failed"
        }
      }
    }

    api.realtime_event {
      channel = "new-contacts-created"
      data = $raw_body|json_encode
      auth_table = "0"
      auth_id = ""
    }
  }

  response = $raw_body
  tags = ["nylas - webhook", "maybe depreciated??"]
}
````
