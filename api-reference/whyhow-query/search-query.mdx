<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>
<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>
<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

<ParamField body="query" type="string">
  The natural language query string.
</ParamField>
<ParamField body="node_names" type="string[]">
  List of node names to filter the query against.
</ParamField>
<ParamField body="is_secondary" type="boolean">
  Whether this is a secondary (follow-up) query.
</ParamField>
<ParamField body="whyhow_query_id" type="integer">
  The ID of an existing query to follow up on.
</ParamField>

<ResponseField name="result" type="object">
  The query response object containing matched results.
</ResponseField>

<RequestExample>
```bash cURL

  -d '{
    "query": "Who do I know at Google?",
```
