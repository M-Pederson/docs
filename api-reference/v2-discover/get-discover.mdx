---

## Overview

Returns a single discover query record by its ID.

## Headers

<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>

<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

## Path Parameters

<ParamField path="discover_id" type="integer" required>
  The ID of the discover record to retrieve.
</ParamField>

## Response

<ResponseField name="id" type="integer">
  The discover record ID.
</ResponseField>

<ResponseField name="created_at" type="timestamp">
  When the record was created.
</ResponseField>

<ResponseField name="updated_at" type="timestamp">
  When the record was last updated.
</ResponseField>

<ResponseField name="user_id" type="integer">
  The ID of the user who created the discover query.
</ResponseField>

<ResponseField name="query" type="string">
  The discover query string.
</ResponseField>

<ResponseField name="edges" type="object">
  Graph edges data, or null.
</ResponseField>

<ResponseField name="answer" type="string">
  The generated answer, or null if still processing.
</ResponseField>

<ResponseField name="archived" type="boolean">
  Whether the discover record is archived.
</ResponseField>

<ResponseField name="processing" type="boolean">
  Whether the discover query is still being processed.
</ResponseField>
