## mvp/node/add-company-node — #2602

Creates the initial FalkorDB graph node for a new company. The node label is dynamic based on type (Company, School, VC_Firm, Organization).

```text
mvp/node/add-company-node — #2602
```

// Version v4 — LLM enrich properties (business_model, primary_product_*, other_product_*, target_customers, acquired_*) + created_at timestamp
CREATE (p:{{$var.nodeLabel}} {
  uuid: randomUUID(),
  created_at: timestamp(),
  master_company_id: {{$var.nodeMcId}},
  display_label: "{{$var.labelType}}",
  name: "{{$var.masterCompany.company_name}}",
  company_domain: "{{$var.masterCompany.company_domain}}",
  {% if $var.alternativeDomains is not null and $var.alternativeDomains|length > 0 %}

  {% endif %}

  name_embedding: vecf32({{$var.embeddingArray}}),
  {% if $var.geminiVecString is not null %}
  embeddings: vecf32({{$var.geminiVecString}}),
  {% endif %}
  {% if $var.llmBusinessModel is not null %}
  business_model: {{$var.llmBusinessModel}},
  {% endif %}
  {% if $var.llmPrimaryProductName is not null %}
  primary_product_name: {{$var.llmPrimaryProductName}},
  {% endif %}
  {% if $var.llmPrimaryProductCategory is not null %}
  primary_product_category: {{$var.llmPrimaryProductCategory}},
  {% endif %}
  {% if $var.llmOtherProductNames is not null %}

  {% endif %}
  {% if $var.llmOtherProductCategories is not null %}
  other_product_categories: {{$var.llmOtherProductCategories}},
  {% endif %}
  {% if $var.llmTargetCustomers is not null %}
  target_customers: {{$var.llmTargetCustomers}},
  {% endif %}
  {% if $var.llmIsAcquired %}

  {% endif %}

  acquired_by: {{$var.llmAcquiredBy}},

  {% if $var.llmAcquiredTicker is not null %}

## mvp/node/update-company-node — #4659

Updates all company properties and embedding on the graph node.

```text
mvp/node/update-company-node — #4659
```

```cypher
// Version v2 — updated_at switched to FalkorDB timestamp() (ms since epoch) for consistency with created_at
MATCH (n:Entity {uuid: "{{$var.masterCompany.node_uuid}}"})
SET {{$var.propertyString}},
    n.name_embedding = vecf32({{$var.embeddingArray}}),{% if $var.geminiVecString is not null %}
    n.embeddings = vecf32({{$var.geminiVecString}}),{% endif %}
    n.updated_at = timestamp()
RETURN n
```

### MERGE Country Node

```cypher
// Version v1
MERGE (p:Entity:Country {name: "{{$var.countryRecord.country_name}}"})
ON CREATE SET
  p.uuid = randomUUID(),
  p.display_label = "Country",
  p.created_at = timestamp(),
  p.updated_at = timestamp()
  {% if $var.countryRecord.flag %}, p.flag = "{{$var.countryRecord.flag}}"{% endif %}
  {% if $var.countryRecord.summary %}, p.summary = "{{$var.countryRecord.summary}}"{% endif %}
  {% if $var.countryRecord.alpha_2_code %}, p.alpha_2_code = "{{$var.countryRecord.alpha_2_code}}"{% endif %}
  {% if $var.countryRecord.population %}, p.population = {{$var.countryRecord.population}}{% endif %}
  {% if $var.countryRecord.capital_city %}, p.capital_city = "{{$var.countryRecord.capital_city}}"{% endif %}
  {% if $var.countryRecord.currency_code %}, p.currency_code = "{{$var.countryRecord.currency_code}}"{% endif %}
  {% if $var.countryRecord.currency_name %}, p.currency_name = "{{$var.countryRecord.currency_name}}"{% endif %}
  {% if $var.embeddingArray %}, p.name_embedding = vecf32({{$var.embeddingArray}}){% endif %}
ON MATCH SET
  p.updated_at = timestamp()
RETURN p
```

### MERGE Location Edges (City + Region + Country)

```cypher
// Version v1
MATCH (person:Entity {uuid: "{{$var.masterPerson.node_uuid}}"})
MATCH (city:Entity {uuid: "{{$var.cityRecord.node_uuid}}"})
MATCH (region:Entity {uuid: "{{$var.regionRecord.node_uuid}}"})
MATCH (country:Entity {uuid: "{{$var.countryRecord.node_uuid}}"})

MERGE (person)-[relates1:LOCATED_IN]->(city)
  ON CREATE SET
    relates1.uuid = randomUuid(),
    relates1.description = person.name + ' is located in ' + city.name,
    relates1.weight = 70,
    relates1.name = "LOCATED_IN",
    relates1.valid_from = "{{'now'|date('Y-m')}}"
  SET relates1.created_at = coalesce(relates1.created_at, timestamp()), relates1.updated_at = timestamp()

WITH person, city, region, country

MERGE (city)-[relates2:IN_REGION]->(region)
  ON CREATE SET
    relates2.uuid = randomUuid(),
    relates2.description = city.name + ' is a city located in ' + region.name,
    relates2.weight = 25,
    relates2.name = "IN_REGION"
  SET relates2.created_at = coalesce(relates2.created_at, timestamp()), relates2.updated_at = timestamp()

MERGE (region)-[relates3:IN_COUNTRY]->(country)
  ON CREATE SET
    relates3.uuid = randomUuid(),
    relates3.description = CASE
      WHEN country.name = 'United States'
        THEN region.name + ' is a state located in the United States'
      WHEN country.name ENDS WITH 's'
        THEN region.name + ' is a region located in the ' + country.name
      ELSE region.name + ' is a region located in ' + country.name
    END,
    relates3.weight = 50,
    relates3.name = "IN_COUNTRY"
  SET relates3.created_at = coalesce(relates3.created_at, timestamp()), relates3.updated_at = timestamp()

WITH person, city, region, country

OPTIONAL MATCH (person)-[relates1:LOCATED_IN]->(city)
OPTIONAL MATCH (city)-[relates2:IN_REGION]->(region)
OPTIONAL MATCH (region)-[relates3:IN_COUNTRY]->(country)

RETURN [relates1.uuid, relates2.uuid, relates3.uuid] AS edge_ids
```
