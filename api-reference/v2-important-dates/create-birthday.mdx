---
title: "Create Birthday"
api: "POST https://xh2o-yths-38lt.n7c.xano.io/api:mGnfV_UB/birthdays"
description: "Add a birthday record for a person."
---

<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>
<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>
<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

<ParamField body="master_person_id" type="integer" required>
  The ID of the master person.
</ParamField>
<ParamField body="date" type="string" required>
  The date. Accepts full date (`YYYY-MM-DD`) or month and day only (`MM-DD`).
</ParamField>
<ParamField body="birthplace_location" type="string">
  Optional birthplace location.
</ParamField>

<ResponseField name="id" type="integer">
  The birthday record ID.
</ResponseField>
<ResponseField name="created_at" type="number">
  Creation timestamp.
</ResponseField>
<ResponseField name="master_person_id" type="integer">
  The associated person ID.
</ResponseField>
<ResponseField name="birthdate" type="string">
  The full birthdate if provided.
</ResponseField>
<ResponseField name="birthday_month" type="integer">
  The birth month.
</ResponseField>
<ResponseField name="birthday_day" type="integer">
  The birth day.
</ResponseField>
<ResponseField name="birthplace_location" type="string">
  The birthplace location.
</ResponseField>

<RequestExample>
```bash cURL
curl -X POST \
  https://xh2o-yths-38lt.n7c.xano.io/api:mGnfV_UB/birthdays \
  -H "Content-Type: application/json" \
  -H "X-Data-Source: live" \
  -H "X-Branch: v1" \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -d '{
    "master_person_id": 42,
    "date": "1990-05-15",
    "birthplace_location": "New York, NY"
  }'
```
</RequestExample>
