---
title: "Update Email Preferences"
api: "PATCH https://xh2o-yths-38lt.n7c.xano.io/api:U2vqGzGG/emails"
description: "Enables or disables emails from user seeing them in the widget."
---

<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>

<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

### Body Parameters

<ParamField body="email_address" type="string" required>
  The email address to update preferences for (email format).
</ParamField>

<ParamField body="email_widget_preference" type="boolean">
  Whether to show this email in the widget.
</ParamField>

<ParamField body="ai_enabled" type="boolean">
  Whether AI features are enabled for this email. Nullable.
</ParamField>

<ParamField body="nylas_grant_id" type="integer">
  The Nylas grant ID associated with this email.
</ParamField>

### Response

<ResponseField name="id" type="integer">
  Unique identifier of the email record.
</ResponseField>

<ResponseField name="created_at" type="timestamp">
  Timestamp when the record was created.
</ResponseField>

<ResponseField name="updated_at" type="timestamp">
  Timestamp when the record was last updated.
</ResponseField>

<ResponseField name="user_id" type="integer">
  The user ID associated with this email.
</ResponseField>

<ResponseField name="grant_id" type="string">
  The Nylas grant identifier.
</ResponseField>

<ResponseField name="email_account" type="string">
  The email account address.
</ResponseField>

<ResponseField name="provider" type="string">
  The email provider (`google`, `microsoft`, `icloud`, `imap`, `yahoo`).
</ResponseField>

<ResponseField name="email_type" type="string">
  The type of email account (`work`, `personal`, `other`).
</ResponseField>

<ResponseField name="show_in_email_widget" type="boolean">
  Whether this email is displayed in the email widget.
</ResponseField>
