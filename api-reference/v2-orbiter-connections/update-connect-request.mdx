---
title: "Update Connect Request"
api: "PUT https://xh2o-yths-38lt.n7c.xano.io/api:8Pr6-Q01/orbiter-connect-requests/{orbiter_connect_request_id}"
description: "Update an orbiter connect request record (accept, decline, block, or clear)."
---

<ParamField header="X-Data-Source" type="string" required>
  Identifies the data source for the request.
</ParamField>

<ParamField header="X-Branch" type="string" required>
  The Xano branch to target (e.g. `v1`, `staging`, `dev`).
</ParamField>

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from login or signup.
</ParamField>

### Path Parameters

<ParamField path="orbiter_connect_request_id" type="integer" required>
  The ID of the connect request to update.
</ParamField>

### Body Parameters

<ParamField body="accepted" type="boolean">
  Set to `true` to accept the connection request.
</ParamField>

<ParamField body="declined" type="boolean">
  Set to `true` to decline the connection request.
</ParamField>

<ParamField body="cleared" type="boolean">
  Set to `true` to clear the connection request. Nullable.
</ParamField>

<ParamField body="blocked" type="boolean">
  Set to `true` to block the requesting user.
</ParamField>

<ParamField body="accept_message" type="string">
  Optional message to include with the acceptance. Nullable.
</ParamField>

### Response

<ResponseField name="id" type="integer">
  Unique identifier of the connect request.
</ResponseField>

<ResponseField name="created_at" type="timestamp">
  Timestamp when the request was created.
</ResponseField>

<ResponseField name="updated_at" type="timestamp">
  Timestamp when the request was last updated.
</ResponseField>

<ResponseField name="user_from_id" type="integer">
  User ID who sent the request.
</ResponseField>

<ResponseField name="user_to_id" type="integer">
  User ID who received the request.
</ResponseField>

<ResponseField name="request_message" type="string">
  The original request message.
</ResponseField>

<ResponseField name="declined" type="boolean">
  Whether the request was declined.
</ResponseField>

<ResponseField name="from_blocked" type="boolean">
  Whether the requesting user is blocked.
</ResponseField>

<ResponseField name="connected_on" type="string">
  Date the connection was established.
</ResponseField>

<ResponseField name="accept_message" type="string">
  Message included with the acceptance.
</ResponseField>

<ResponseField name="cleared" type="boolean">
  Whether the request was cleared.
</ResponseField>

  }'
