## nylas-webhook-event-deleted — #2165

Triggered by the Nylas `event.deleted` webhook when a calendar event is cancelled. Looks up the Nylas grant, calendar, and event records, then creates a "Meeting has been deleted" activity for every participant's `my_person` record. Soft-deletes both the `event` and `master_event` records by setting `deleted_at`.

// Creates an activity of type 'event_deleted' and removes the event.
function "mvp/nylas-webhook/nylas-webhook-event-deleted" {
  input {
    text grant_id filters=trim
    json event
  }

  stack {
    // TEMP!!!!
    group {
      stack {
        return {
          value = ""
        }
      }
    }

    db.get nylas_grant {
      field_name = "grant_id"
      field_value = $input.grant_id
      output = ["id", "user_id", "grant_id", "email_account"]
    } as $nylasGrant

    db.get calendar {
      field_name = "external_id"
      field_value = $input.event.calendar_id
      output = ["id", "external_id", "grant_id", "name"]
    } as $calendar

    db.query event {
      where = $db.event.external_id == $input.event.id && $db.event.calendar_id == $calendar.id
      return = {type: "single"}
    } as $dbEvent

    conditional {
      if ($nylasGrant == null || $calendar == null || $dbEvent == null) {
        return {
          value = "Either of Nylas Grant,Event or Calendar not found"
        }
      }
    }

    db.query activity {
      where = $db.activity.event_id == $dbEvent.id
      return = {type: "list"}
    } as $activities

    var $myPersonIds {
      value = $activities|get:"my_person_id":null
    }

    db.query my_person {
      where = $db.my_person.user_id == $nylasGrant.user_id && $db.my_person.id in $myPersonIds
      return = {type: "list"}
      output = [
        "id"
        "created_at"
        "updated_at"
        "user_id"
        "master_person_id"
        "full_name"
        "first_name"
        "middle_name"
        "last_name"
        "suffix"
        "nickname"
        "birthdate"
        "keep_in_touch"
        "status_orbiter"
        "status_connected"
        "last_activity_at"
        "last_activity_id"
        "email_address_id"
      ]
    } as $myPersons

    foreach ($myPersons) {
      each as $myPerson {
        db.add activity {
          data = {
            created_at        : "now"
            my_person_id      : $myPerson.id
            activity_type_id  : 18
            card_descriptor   : "Meeting has been deleted"
            activity_timestamp: now
            subject           : $input.event|get:"title":null
            grant_id          : $input.grant_id
            event_id          : $dbEvent.id
            data_source_id    : 21
            status            : "deleted"
          }
        } as $activity

        db.edit my_person {
          field_name = "id"
          field_value = $myPerson.id
          data = {
            last_activity_at: $activity.activity_timestamp
            last_activity_id: $activity.id
          }
        }
      }
    }

    db.query master_event {
      where = $db.master_event.external_id == $input.event.id && $db.master_event.calendar_id == $calendar.id
      return = {type: "list"}
      output = ["id"]
    } as $masterEvent

    conditional {
      if (($masterEvent|is_empty) == false) {
        db.edit master_event {
          field_name = "id"
          field_value = $masterEvent.id
          data = {deleted_at: now}
        }
      }
    }

    db.edit event {
      field_name = "id"
      field_value = $dbEvent.id
      data = {deleted_at: now}
      output = ["id", "title", "description", "deleted_at"]
    } as $deletedEvent
  }

  response = $deletedEvent
  tags = ["nylas - webhook"]
}
