## mvp/funding/add-fundable-deal-node (function #12701)

### MERGE Funding Round Node

```cypher
// Version v1

ON CREATE SET
  fr.visibility = true,
  fr.display_label = 'Funding_Round',
  fr.name = '{{$var.nodeName}}'
  {% if $var.organization.name is not empty %},
    fr.company_name = '{{$var.organization.name}}'{% endif %}
  {% if $var.fundableDeal.financing_type is not empty %},
    fr.funding_type = '{{$var.fundableDeal.financing_type}}'{% endif %}
  {% if $var.formattedDate is not empty %},
    fr.announced_date = '{{$var.formattedDate}}'{% endif %}
  {% if $var.sizeUsdActual > 0 %},
    fr.size_usd = {{$var.sizeUsdActual}}{% endif %}
  {% if $var.sizeActual > 0 %},
    fr.size = {{$var.sizeActual}}{% endif %}
  {% if $var.fundableDeal.currency is not empty %},
    fr.currency = '{{$var.fundableDeal.currency}}'{% endif %}
  {% if $var.fundableDeal.short_description is not empty %},
    fr.short_description = '{{$var.fundableDeal.short_description}}'{% endif %}
  {% if $var.fundableDeal.long_description is not empty %},
    fr.long_description = '{{$var.fundableDeal.long_description}}'{% endif %}
  {% if $var.angelCount > 0 %},
    fr.number_of_angel_investors = {{$var.angelCount}}{% endif %}
  {% if $var.institutionalCount > 0 %},
    fr.number_of_institutional_investors = {{$var.institutionalCount}}{% endif %}
  {% if $var.totalInvestorCount > 0 %},
    fr.total_number_of_investors = {{$var.totalInvestorCount}}{% endif %}
  , fr.name_embedding = vecf32([{{$var.embeddingArray|join(',')}}])
ON MATCH SET
  fr.visibility = true,
  fr.display_label = 'Funding_Round',
  fr.name = '{{$var.nodeName}}'
  {% if $var.organization.name is not empty %}, fr.company_name = '{{$var.organization.name}}'{% endif %}
  {% if $var.fundableDeal.financing_type is not empty %}, fr.funding_type = '{{$var.fundableDeal.financing_type}}'{% endif %}
  {% if $var.formattedDate is not empty %}, fr.announced_date = '{{$var.formattedDate}}'{% endif %}
  {% if $var.sizeUsdActual > 0 %}, fr.size_usd = {{$var.sizeUsdActual}}{% endif %}
  {% if $var.sizeActual > 0 %}, fr.size = {{$var.sizeActual}}{% endif %}
  {% if $var.fundableDeal.currency is not empty %}, fr.currency = '{{$var.fundableDeal.currency}}'{% endif %}
  {% if $var.fundableDeal.short_description is not empty %}, fr.short_description = '{{$var.fundableDeal.short_description}}'{% endif %}
  {% if $var.fundableDeal.long_description is not empty %}, fr.long_description = '{{$var.fundableDeal.long_description}}'{% endif %}

  , fr.name_embedding = vecf32([{{$var.embeddingArray|join(',')}}])
```

### MERGE Company RAISED Edge

```cypher
// Version v1
MATCH (c:Company {uuid: '{{$var.organization.master_company_node_uuid}}'})
MATCH (fr:Funding_Round {uuid: '{{$var.dealId}}'})
MERGE (c)-[raised:RAISED]->(fr)
SET raised.weight = 10
  {% if $var.formattedDate is not empty %},
    raised.announced_date = '{{$var.formattedDate}}'{% endif %}
  {% if $var.organization.name is not empty %},
    raised.company_name = '{{$var.organization.name}}'{% endif %}
  {% if $var.fundableDeal.short_description is not empty %},
    raised.description = '{{$var.fundableDeal.short_description}}'{% endif %}
RETURN c.name AS company, fr.uuid AS node_uuid
```

## mvp/fundable/resolve-investors-edges (function #12702)

Resolves all investor nodes for a deal, then creates investor-to-funding-round edges. Handles institutional investors (VC firms), angel investors, and investment partners.

### Fix Domain-Slug Company Name

When an investor company node has a domain-slug name (e.g. `acme_com`), update it to the proper name from Fundable.

```cypher
// Version v1
MATCH (c:Company {uuid: '{{$var.investorNodeUuid}}'})
SET c.company_name = '{{$var.safeInvestorOrgName}}',
    c.name = '{{$var.safeInvestorOrgName}}'
RETURN c.company_name
```

### Institutional Investor — Lead (LEAD_INVESTED_IN)

```cypher
// Version v1
MATCH (vc:Company {uuid: '{{$var.investorNodeUuid}}'})
SET vc:VC_Firm
WITH vc
MATCH (fr:Funding_Round {uuid: '{{$var.deal.funding_round_node_uuid}}'})
MERGE (vc)-[r:LEAD_INVESTED_IN]->(fr)
SET r.uuid = '{{$var.investorNodeUuid}}-{{$var.deal.funding_round_node_uuid}}',
    r.name = NULL,
    r.description = '{{$var.safeInvestorOrgName}} led the
      {{$var.safeFundedCoName}} {{$var.safeFinancingType}} funding round.',
    r.is_lead_investor = true,
    r.weight = 30
RETURN vc.name AS investor, fr.uuid AS funding_round,
       'LEAD_INVESTED_IN' AS rel_type
```

### Institutional Investor — INVESTED_IN / FOLLOW_ON_INVESTED_IN

Checks if the VC has invested in earlier rounds of the same company. If so, creates a FOLLOW_ON_INVESTED_IN edge instead of INVESTED_IN.

```cypher
// Version v1
MATCH (vc:Company {uuid: '{{$var.investorNodeUuid}}'})
SET vc:VC_Firm
WITH vc
MATCH (fr:Funding_Round {uuid: '{{$var.deal.funding_round_node_uuid}}'})
With vc, fr
MATCH (fundedCompany:Company {uuid: '{{$var.fundedCompanyOrg.master_company_node_uuid}}'})
OPTIONAL MATCH (fundedCompany)-[:RAISED]->(earlierRound:Funding_Round)
  <-[earlierInvestment]-(vc)
WHERE earlierRound.announced_date < '{{$var.formattedDate}}'
  AND type(earlierInvestment) IN
    ['LEAD_INVESTED_IN', 'INVESTED_IN', 'FOLLOW_ON_INVESTED_IN']
WITH vc, fr, COUNT(earlierRound) > 0 AS hasEarlierInvestment

FOREACH (_ IN CASE WHEN NOT hasEarlierInvestment THEN [1] ELSE [] END |
  MERGE (vc)-[r:INVESTED_IN]->(fr)
  SET r.uuid = '{{$var.investorNodeUuid}}-{{$var.deal.funding_round_node_uuid}}',
      r.name = NULL,
      r.description = '{{$var.safeInvestorOrgName}} invested in the
        {{$var.safeFundedCoName}} {{$var.safeFinancingType}} funding round.',
      r.is_lead_investor = false,
      r.weight = 20
)

FOREACH (_ IN CASE WHEN hasEarlierInvestment THEN [1] ELSE [] END |
  MERGE (vc)-[r:FOLLOW_ON_INVESTED_IN]->(fr)
  SET r.uuid = '{{$var.investorNodeUuid}}-{{$var.deal.funding_round_node_uuid}}',
      r.name = NULL,
      r.description = '{{$var.safeInvestorOrgName}} made a follow-on investment
        in the {{$var.safeFundedCoName}} {{$var.safeFinancingType}} funding round.',
      r.is_lead_investor = false,
      r.weight = 25
)

With vc, fr
MATCH (vc)-[r]->(fr)
WHERE type(r) IN ['INVESTED_IN', 'FOLLOW_ON_INVESTED_IN']
RETURN vc.name AS investor, fr.uuid AS funding_round, type(r) AS rel_type
```

### Investment Partner (INVESTMENT_PARTNER_IN + INVESTMENT_PARTNER_AT)

Links an individual person (partner at a VC firm) to both the funding round and their firm for a specific deal.

```cypher
// Version v1
MATCH (p:Person {uuid: '{{$var.partnerNodeUuid}}'})
MATCH (fr:Funding_Round {uuid: '{{$var.deal.funding_round_node_uuid}}'})
MATCH (vc:Company {uuid: '{{$var.investorNodeUuid}}'})

MERGE (p)-[dr:INVESTMENT_PARTNER_IN]->(fr)
SET dr.uuid = '{{$var.partnerNodeUuid}}-{{$var.deal.funding_round_node_uuid}}',
    dr.description = '{{$var.safePartnerName}} was an investment partner in
      the {{$var.safeFundedCoName}} {{$var.safeFinancingType}} funding round,
      representing {{$var.safeInvestorOrgName}}.',
    dr.weight = 15

MERGE (p)-[dv:INVESTMENT_PARTNER_AT]->(vc)
SET dv.uuid = '{{$var.partnerNodeUuid}}-{{$var.investorNodeUuid}}-deal_partner',
    dv.description = '{{$var.safePartnerName}} represented
      {{$var.safeInvestorOrgName}} as an investment partner in the
      {{$var.safeFundedCoName}} {{$var.safeFinancingType}} funding round.',
    dv.weight = 20

RETURN p.name AS partner, fr.uuid AS funding_round, vc.name AS vc_firm
```
