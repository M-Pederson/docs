## nylas-webhook/message-read — API #858

Triggered by the Nylas `message.updated` webhook when an email's read status changes. Validates the webhook signature using HMAC-SHA256, looks up the corresponding `activity` record by message ID, updates its `unread` flag, and sends a real-time notification via the `nylas-send-authenticated-realtime` channel.

Note: This is an **API endpoint** (not a function) that handles the webhook directly, including challenge verification for Nylas webhook registration.

```text
nylas-webhook/message-read — API #858
```

<Accordion title="XanoScript — click to expand">
````xanoscript
// Nylas sends message.updated data to this endpoint
query "nylas-webhook/message-read" verb=POST {
  api_group = "Nylas"

  input {
    text challenge? filters=trim
  }

  stack {
    conditional {
      if ($input.challenge != null) {
        util.set_header {
          value = "Content-Type: text/html; charset=UTF-8"
          duplicates = "replace"
        }

        return {
          value = $input.challenge|escape
        }
      }
    }

    util.get_raw_input {
      encoding = "none"
      exclude_middleware = false
    } as $raw_body

    // Check data source from client id from nylas
    conditional {
      if ($raw_body.data.application_id == $env.nylas_client_id) {
        db.set_datasource {
          value = "staging"
        }
      }

      else {
        db.set_datasource {
          value = "live"
        }
      }
    }

    function.run "mvp/nylas/get-nylas-keys" as $keys
    var $hash {
      value = $raw_body
        |hmac_sha256:$keys.nylas_webhook_message_read_secret:false
    }

    // Check if data is coming from nylas
    conditional {
      if ($hash != $env.$http_headers["X-Nylas-Signature"]) {
        return {
          value = "signature failed"
        }
      }
    }

    util.post_process {
      stack {
        util.get_raw_input {
          encoding = "json"
          exclude_middleware = false
        } as $body

        var $messageId {
          value = $body|get:"data.object.id":null
        }

        var $isUnread {
          value = $body|get:"data.object.unread":null
        }

        db.get activity {
          field_name = "message_id"
          field_value = $messageId
          output = ["id"]
        } as $messageActivity

        conditional {
          if (($messageActivity|is_empty) == false) {
            db.edit activity {
              field_name = "id"
              field_value = $messageActivity.id
              data = {unread: $isUnread}
              output = ["id", "message_id"]
            } as $activity

            var $grantId {
              value = $body|get:"data.object.grant_id":null
            }

            function.run "nylas/nylas-send-authenticated-realtime" {
              input = {
                message : "Email has been read"
                grant_id: $grantId
                channel : "email_read"
              }
            }
          }
        }
      }
    }
  }

  response = "success"
  tags = ["nylas - webhook"]
}
````
